# 아이템4 구조적 타이핑에 익숙해지기

### Q1. 다음 코드에서 어떤 오류가 발생하는지와 이유를 설명해주세요. (경민)

```js
interface Vector3D {
  x: number;
  y: number;
  z: number;
}

function calculateLengthL1(v: Vector3D) {
  let length = 0;
  for (const axis of Object.keys(v)) {
    const coord = v[axis];
    length += Math.abs(coord);
  }
  return length;
}
```
